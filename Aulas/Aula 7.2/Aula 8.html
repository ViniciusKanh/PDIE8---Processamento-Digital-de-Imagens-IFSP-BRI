Parmetros para a subtrao de fundo

In[]:
alpha = 0.95
theta = 0.1
background = None
In[]:
while True:
    ret, frame = video_capture.read()
    if not ret:
        break

    # Converter o frame para escala de cinza
    gray_frame = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)

    if background is None:
        background = gray_frame.astype(float)
        continue

    # Atualizar o modelo de fundo com suavizao exponencial
    background = alpha * background + (1 - alpha) * gray_frame

    # Calcular a diferena entre o frame atual e o fundo
    diff_frame = np.abs(gray_frame - background)
    thresh_frame = (diff_frame > theta * 255).astype(np.uint8)

    # Gravar o frame de sada
    out.write(cv2.cvtColor(thresh_frame * 255, cv2.COLOR_GRAY2BGR))

    if cv2.waitKey(1) & 0xFF == ord('q'):
        break

# Liberar recursos
video_capture.release()
out.release()
cv2.destroyAllWindows()
